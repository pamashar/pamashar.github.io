This


this - is a keyword to define what scope is relevant for code block



class User() {
  getThisArrowClass = () => console.log(this);
  getThisRegularClass() { console.log(this); };
}

const user = {
  name: 'Abc',
  getThisArrow: () => console.log(this),
  getThisRegular: function () {
    console.log(this);
  },
}

user.getThisArrow() // Window object. Where function executed
user.getThisRegular() // user object. Where function defined

(new User()).getThisArrow() // User instance. Constructor set this
(new User()).getThisRegular() // User instance. Constructor set this
